int led = 10;         // O Pino ao qual o LED está conectado
int brilho = 0;      // O nível de brilho do LED. 0 é o mínimo e 255 é o máximo

// O bloco SETUP onde o código executa
void setup() {
  
  // Use pinMode para que o led esteja no modo OUTPUT/Saída
  pinMode(led, OUTPUT);
}

// O bloco LOOP que faz o código se repetir infinitamente
void loop() {
  // identifica a entrada o pino analógico sendo 0:
  int valor_sensor = analogRead(A0);

  // Converte a leitura analógica (que vai de 0 - 1023) para um voltagem entre (0v - 5V):
  float voltagem = valor_sensor * (5.0 / 1023.0);
  //exibe o valor que está sendo lido

  // Define o nível de brilho do LED
  analogWrite(led, brilho);

  if (voltagem == 0){

    brilho = 0;
    
  }else if (voltagem > 0 && voltagem <= 1){
          
    brilho = 15;
          
  }else if (voltagem > 1 && voltagem <= 2){

    brilho = 100;

  }else if (voltagem > 2 && voltagem <= 3){

    brilho = 150;
          
  }else if (voltagem > 3){

    brilho = 250;

  }

  Serial.println(voltagem);
  
  // Atraso/delay o efeito cada mudança em uma quantidade de milisegundos
  // 1000 equivale a 1 segundo e 500 equivale a 0,5 segundo
  delay(100);
}
